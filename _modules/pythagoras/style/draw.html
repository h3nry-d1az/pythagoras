@dataclass
class FontSize(POProperty):
    """
    Font size specifier.

    Attributes:
        size: Font size value.
        unit: Unit for the font size (default: 'pt').
    """

    size: float
    unit: str = "pt"

    def svg(self) -> str:
        return f'font-size="{self.size}"'

    def tikz(self) -> str:
        return rf"font=\fontsize{{{self.size}{self.unit}}}{{{self.size * 1.2:.4f}{self.unit}}}\selectfont"
